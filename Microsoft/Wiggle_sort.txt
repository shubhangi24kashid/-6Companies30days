Wiggle sort

class Solution {
public:
    void wiggleSort(vector<int>& nums) {
        vector<int>sorted=nums;
        sort(sorted.begin(),sorted.end());

        int n=nums.size();
        int mid=(n-1)/2;
        int end=(n-1);

        for(int i=0;i<n;i++){
            if(i%2==0){
                nums[i]=sorted[mid--];
            }
            else{
                nums[i]=sorted[end--];
            }
        }
    }
};
